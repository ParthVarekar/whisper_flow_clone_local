# WhisperFlow runtime dependencies.
#
# The core whisper_flow package is stdlib-only, but the daemon mode requires
# these packages for ASR, mic capture, hotkeys, and the system tray UI.
#
# Install with:
#   pip install -r requirements.txt
#
# Or just run start.bat — it auto-installs any missing packages on first run.

# ASR backend — Moonshine Tiny (27M, English-only, ONNX Runtime, in-process)
# No external binary/server needed. Model auto-downloads on first use (~34MB).
moonshine-voice>=0.1.0

# Microphone capture (cross-platform PortAudio binding)
sounddevice>=0.4.6

# Global hotkey listener (Ctrl+Shift+Space for push-to-talk)
# CRITICAL: without this, the daemon cannot listen for hotkeys and dictation
# will not work at all.
pynput>=1.7.6

# System tray icon (daemon UI)
pystray>=0.19.5

# Image support for the tray icon (required by pystray)
Pillow>=10.0.0

# Numerical arrays (required by moonshine-voice + sounddevice + audio processing)
numpy>=1.24.0

# Clipboard fallback for text insertion (used if ctypes SendInput fails on Windows)
pyperclip>=1.8.2

# TOML config parsing on Python 3.8-3.10 (3.11+ has stdlib tomllib)
tomli; python_version < "3.11"
